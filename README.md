code:
```
let name = prompt("Enter your name");
let birthYear = prompt("Enter your birth year");
let job = prompt("Do you have a job? Write Yes or No")
let age = 2025 - birthYear;
  if (age >= 18 && job == "Yes") {
    alert("Wellcome!");
  }
  else {
    let message = prompt("What do you do here?");
  }
```
